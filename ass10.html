import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.*;

public class BMICalculator extends JFrame {
    private JComboBox<String> genderComboBox;
    private JTextField heightTextField;
    private JTextField weightTextField;
    private JLabel resultLabel;

    public BMICalculator() {
        setTitle("BMI Calculator");
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setSize(350, 250);
        setLayout(new GridBagLayout());

        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(5, 5, 5, 5);

        JLabel genderLabel = new JLabel("Select Gender:");
        gbc.gridx = 0;
        gbc.gridy = 0;
        gbc.gridwidth = 1;
        add(genderLabel, gbc);

        JLabel heightLabel = new JLabel("Enter Height (cm):");
        gbc.gridx = 0;
        gbc.gridy = 1;
        gbc.gridwidth = 1;
        add(heightLabel, gbc);

        JLabel weightLabel = new JLabel("Enter Weight (kg):");
        gbc.gridx = 0;
        gbc.gridy = 2;
        gbc.gridwidth = 1;
        add(weightLabel, gbc);

        JButton calculateButton = new JButton("Calculate BMI");
        calculateButton.setPreferredSize(new Dimension(150, 30));
        calculateButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                calculateBMI();
            }
        });
        gbc.gridx = 0;
        gbc.gridy = 3;
        gbc.gridwidth = 1;
        add(calculateButton, gbc);

        String[] genders = {"Male", "Female"};
        genderComboBox = new JComboBox<>(genders);
        gbc.gridx = 1;
        gbc.gridy = 0;
        gbc.gridwidth = 1;
        add(genderComboBox, gbc);

        heightTextField = new JTextField(10);
        gbc.gridx = 1;
        gbc.gridy = 1;
        gbc.gridwidth = 1;
        add(heightTextField, gbc);

        weightTextField = new JTextField(10);
        gbc.gridx = 1;
        gbc.gridy = 2;
        gbc.gridwidth = 1;
        add(weightTextField, gbc);

        resultLabel = new JLabel("");
        gbc.gridx = 0;
        gbc.gridy = 4;
        gbc.gridwidth = 2;
        add(resultLabel, gbc);

        setVisible(true);
    }

    private void calculateBMI() {
        try {
            String gender = (String) genderComboBox.getSelectedItem();
            double height = Double.parseDouble(heightTextField.getText());
            double weight = Double.parseDouble(weightTextField.getText());

            double bmi = calculateBMI(height, weight);

            String result = "Your BMI is: " + bmi + " - ";
            if (gender.equals("Male")) {
                result += getBMIResultForMale(bmi);
            } else if (gender.equals("Female")) {
                result += getBMIResultForFemale(bmi);
            }
            resultLabel.setText(result);

            if (gender.equals("Male") && bmi >= 24.9) {
                resultLabel.setForeground(Color.RED);
            } else if (gender.equals("Female") && bmi >= 23.9) {
                resultLabel.setForeground(Color.YELLOW);
            } else {
            resultLabel.setForeground(Color.BLACK);
            }
    } catch (NumberFormatException e) {
        resultLabel.setText("Please enter valid numbers for height and weight.");
        resultLabel.setForeground(Color.BLACK);
    }
}

    private double calculateBMI(double height, double weight) {
        return weight / Math.pow(height / 100, 2);
}

    private String getBMIResultForMale(double bmi) {
        if (bmi < 18.5) {
            return "Underweight";
        } else if (bmi < 24.9) {
            return "Normal weight";
        } else {
            return "Overweight";
        }
}

    private String getBMIResultForFemale(double bmi) {
        if (bmi < 19) {
            return "Underweight";
        } else if (bmi < 23.9) {
            return "Normal weight";
        } else {
            return "Overweight";
        }
}

    public static void main(String[] args) {
        new BMICalculator();
}

import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.*;

public class KedaiSaya extends JFrame {
    private JButton kiraButton;
    private JLabel jumlahBayaranLabel;
    private JList<String> itemList;

    public KedaiSaya() {
        setTitle("Kedai Saya");
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setSize(350, 300);
        setLayout(new GridLayout(3, 1));

        String[] items = {
            "Kopi O (RM 2.50)",
            "Kopi Ping (RM 3.20)",
            "Milo Kahwin (RM 3.90)",
            "Milo Ping (RM 3.80)",
            "Teh Tarik (RM 3.50)"
};

        itemList = new JList<>(items);
        itemList.setSelectionMode(ListSelectionModel.MULTIPLE_INTERVAL_SELECTION);
        itemList.setFixedCellHeight(30);
        itemList.setFixedCellWidth(100);
        itemList.setVisibleRowCount(5);

        JScrollPane scrollPane = new JScrollPane(itemList);

        JLabel listLabel = new JLabel("Pilih barang yang anda mahu beli. Kemudiaan tekan butang Kira.");
        listLabel.setHorizontalAlignment(SwingConstants.CENTER);

        kiraButton = new JButton("Kira");
        kiraButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                calculateTotal();
            }
        });

        jumlahBayaranLabel = new JLabel("Jumlah Bayaran: RM 0.00");

        JPanel itemPanel = new JPanel(new BorderLayout());
        itemPanel.add(listLabel, BorderLayout.NORTH);
        itemPanel.add(scrollPane, BorderLayout.CENTER);

        JPanel bottomPanel = new JPanel();
        bottomPanel.add(kiraButton);
        bottomPanel.add(jumlahBayaranLabel);

        add(itemPanel);
        add(bottomPanel);

        setVisible(true);
}

    private void calculateTotal() {
        double total = 0.0;
        if (itemList.isSelectedIndex(0)) total += 2.50; // Kopi O
        if (itemList.isSelectedIndex(1)) total += 3.20; // Kopi Ping
        if (itemList.isSelectedIndex(2)) total += 3.90; // Milo Kahwin
        if (itemList.isSelectedIndex(3)) total += 3.80; // Milo Ping
        if (itemList.isSelectedIndex(4)) total += 3.50; // Teh Tarik

        jumlahBayaranLabel.setText("Jumlah Bayaran: RM " + String.format("%.2f", total));
}

    public static void main(String[] args) {
        new KedaiSaya();
}
