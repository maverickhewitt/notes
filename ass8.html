import java.util.Scanner;

abstract class Student {
    private String name;
    private String studentID;
    private String hireDate;

    public Student(String name, String studentID, String hireDate) {
        this.name = name;
        this.studentID = studentID;
        this.hireDate = hireDate;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getStudentID() {
        return studentID;
    }

    public void setStudentID(String studentID) {
        this.studentID = studentID;
    }

    public String getHireDate() {
        return hireDate;
    }

    public void setHireDate(String hireDate) {
        this.hireDate = hireDate;
    }

    public abstract int getRemainingHours();
}

class CompSciStudent extends Student {
    private final int totalMathHours = 20;
    private final int totalCSHours = 40;
    private final int totalGenHours = 60;

    private int completedMathHours;
    private int completedCSHours;
    private int completedGenHours;

    public CompSciStudent(String name, String studentID, String hireDate,
            int completedMathHours, int completedCSHours, int completedGenHours) {
        super(name, studentID, hireDate);
        this.completedMathHours = completedMathHours;
        this.completedCSHours = completedCSHours;
        this.completedGenHours = completedGenHours;
    }

    @Override
    public int getRemainingHours() {
        int remainingMath = totalMathHours - completedMathHours;
        int remainingCS = totalCSHours - completedCSHours;
        int remainingGen = totalGenHours - completedGenHours;

        return remainingMath + remainingCS + remainingGen;
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("\n--- Student Details ---");

        System.out.println("Student ID: " + student.getStudentID());
        System.out.println("Hire Date: " + student.getHireDate());
        System.out.println("Remaining Hours: " + student.getRemainingHours());

        scanner.close();
    }
}

import java.util.Scanner;

interface Shape {
    public double area();
}

class Circle implements Shape {
    private double radius;

    public Circle(double radius) {
        this.radius = radius;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double area() {
        return Math.PI * radius * radius;
    }
}

class Rectangle implements Shape {
    private double height;
    private double width;

    public Rectangle(double height, double width) {
        this.height = height;
        this.width = width;
    }

    public double getHeight() {
        return height;
    }

    public void setHeight(double height) {
        this.height = height;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    @Override
    public double area() {
        return height * width;
    }
}

public class p2Shape {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter the radius of the Circle: ");
        double circleRadius = scanner.nextDouble();

        System.out.print("Enter the height of the Rectangle: ");
        double rectangleHeight = scanner.nextDouble();

        System.out.print("Enter the width of the Rectangle: ");
        double rectangleWidth = scanner.nextDouble();

        Circle circle = new Circle(circleRadius);
        Rectangle rectangle = new Rectangle(rectangleHeight, rectangleWidth);

        System.out.println("Circle Area: " + circle.area());

        System.out.println("Rectangle Area: " + rectangle.area());

        scanner.close();
    }
}
