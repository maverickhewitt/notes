import java.text.DecimalFormat; //required for decimal formatter

public class Lab1_Activity4{
    public static void main(String args[]){
        System.out.printf("%s%n%s%n", "Hello world", "Welcome to Java Programming!");
        double num1 = 2.5432;
        int num2 = 4;
        //formatting decimal using printf
        System.out.printf("Sum of %f and %d is :%.2f%n", num1, num2, num1+num2);

        //formatting decimal using DecimalFormat
        DecimalFormat df = new DecimalFormat("#.##"); //decimal formatter to 2 places
        System.out.println("Division of num1 and num2 is: " + df.format(num1/num2));

        double x, y = 150.50;
        int temp = 5;

        x = y/3-34*temp+120;

        System.out.printf("X equal to: %.2f%n", x);
        System.out.printf("Y equal to: %.2f%n", y);
        }
}

import java.text.DecimalFormat;
import java.util.Scanner;

public class Lab1_Exercise1 {
    public static void main(String[] args) {
        double weight, height, bmi;

        String input_weight = "Enter your weight (kg): ";
        String input_height = "Enter your height (m): ";
        String bmi_result = "";

        Scanner input = new Scanner(System.in);
        System.out.print(input_weight);
        weight = input.nextDouble();
        System.out.print(input_height);
        height = input.nextDouble();

        bmi = weight/(height*height);

        DecimalFormat df = new DecimalFormat("#.##");

        if (bmi < 16.0) {
            bmi_result = "Severely Underweight";
        }
        else if (bmi <= 18.5) {
            bmi_result = "Underweight";
        }
        else if (bmi <= 25.0) {
            bmi_result = "Normal Weight";
        }
        else if (bmi <= 30.0) {
            bmi_result = "Overweight";
        }
        else if (bmi <= 35.0) {
            bmi_result = "Moderately Obese";
        }
  ;    else {
            bmi_result = "Severely Obese";
        }

        System.out.println("Your BMI is " + df.format(bmi) + " which is " + bmi_result);
    }
}
