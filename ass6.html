import javax.swing.JOptionPane;

public class CarpetCostCalculator {
    public static void main(String[] args) {
        double length = Double.parseDouble(JOptionPane.showInputDialog("Enter room length (in feet):"));
        double width = Double.parseDouble(JOptionPane.showInputDialog("Enter room width (in feet):"));
        RoomDimension roomDimension = new RoomDimension(length, width);

        double costPerSquareFoot = Double.parseDouble(JOptionPane.showInputDialog("Enter cost of carpet per square foot:"));

        RoomCarpet roomCarpet = new RoomCarpet(roomDimension, costPerSquareFoot);

        System.out.println("Room Dimensions: " + roomDimension);
        System.out.println("Carpet Cost: $" + String.format("%.2f", roomCarpet.getTotalCost()));
    }
}

class RoomDimension {
    private double length;
    private double width;

    public RoomDimension(double length, double width) {
        this.length = length;
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    // Method to calculate area
    public double getArea() {
        return length * width;
    }

    public String toString() {
        return "Length= " + length + " Width= " + width ;
    }
}

class RoomCarpet {
    private RoomDimension dimension;
    private double costPerSquareFoot;

    public RoomCarpet(RoomDimension dimension, double costPerSquareFoot) {
        this.dimension = dimension;
        this.costPerSquareFoot = costPerSquareFoot;
    }

    public RoomDimension getDimension() {
        return dimension;
    }

    public void setDimension(RoomDimension dimension) {
        this.dimension = dimension;
    }

    public double getCostPerSquareFoot() {
        return costPerSquareFoot;
    }

    public void setCostPerSquareFoot(double costPerSquareFoot) {
        this.costPerSquareFoot = costPerSquareFoot;
    }

    public double getTotalCost() {
        return dimension.getArea() * costPerSquareFoot;
    }

    public String toString() {
        return "Dimension= " + dimension + " Cost Per Square Foot= " + costPerSquareFoot +
        " TotalCost= " + String.format("%.2f", getTotalCost());
    }
}

import javax.swing.JOptionPane;

public class GradingDemo {
    public static void main(String[] args) {
        int quiz1 = Integer.parseInt(JOptionPane.showInputDialog("Enter score for Quiz 1 (out of 10):"));
        int quiz2 = Integer.parseInt(JOptionPane.showInputDialog("Enter score for Quiz 2 (out of 10):"));
        int quiz3 = Integer.parseInt(JOptionPane.showInputDialog("Enter score for Quiz 3 (out of 10):"));
        int midterm = Integer.parseInt(JOptionPane.showInputDialog("Enter score for Midterm (out of 100):"));
        int finalExam = Integer.parseInt(JOptionPane.showInputDialog("Enter score for Final Exam (out of 100):"));
        StudentRecord student = new StudentRecord(quiz1, quiz2, quiz3, midterm, finalExam);

        System.out.println(student);
    }
}

class StudentRecord {
    private int quiz1;
    private int quiz2;
    private int quiz3;
    private int midterm;
    private int finalExam;
    private double overallScore;
    private char letterGrade;

    public StudentRecord(int quiz1, int quiz2, int quiz3, int midterm, int finalExam) {
        this.quiz1 = quiz1;
        this.quiz2 = quiz2;
        this.quiz3 = quiz3;
        this.midterm = midterm;
        this.finalExam = finalExam;
        computeOverallScore();
        computeLetterGrade();
    }

    public int getQuiz1() {
        return quiz1;
    }

    public void setQuiz1(int quiz1) {
        this.quiz1 = quiz1;
    }

    public int getQuiz2() {
        return quiz2;
    }

    public void setQuiz2(int quiz2) {
        this.quiz2 = quiz2;
    }

    public int getQuiz3() {
        return quiz3;
    }

    public void setQuiz3(int quiz3) {
        this.quiz3 = quiz3;
    }

    public int getMidterm() {
        return midterm;
    }

    public void setMidterm(int midterm) {
        this.midterm = midterm;
    }

    public int getFinalExam() {
        return finalExam;
    }

    public void setFinalExam(int finalExam) {
        this.finalExam = finalExam;
    }

    public double getOverallScore() {
        return overallScore;
    }

    public char getLetterGrade() {
        return letterGrade;
    }

    public void computeOverallScore() {
        double quizAverage = ((quiz1 + quiz2 + quiz3) / 30.0) * 25;
        double midtermWeighted = midterm * 0.35;
        double finalExamWeighted = finalExam * 0.40;
        this.overallScore = quizAverage + midtermWeighted + finalExamWeighted;
    }

    public void computeLetterGrade() {
        if (overallScore >= 90) {
            letterGrade = 'A';
        } else if (overallScore >= 80) {
            letterGrade = 'B';
        } else if (overallScore >= 70) {
            letterGrade = 'C';
        } else if (overallScore >= 60) {
            letterGrade = 'D';
        } else {
            letterGrade = 'F';
        }
    }

    public String toString() {
        return "StudentRecord:\n" +
        "Quiz Scores: " + quiz1 + ", " + quiz2 + ", " + quiz3 + "\n" +
        "Midterm: " + midterm + "\n" +
        "Final Exam: " + finalExam + "\n" +
        "Overall Score: " + String.format("%.2f", overallScore) + "\n" +
        "Letter Grade: " + letterGrade;
    }
}
