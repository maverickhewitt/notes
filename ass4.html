import java.util.Scanner;

public class assignment4 {
    public static void main(String[] args) {
        String name;
        int age;
        double height;
        double weight;
        double bmi;
        String bmi_status;

        name = getUser Input("name");
        age = Integer.parseInt(getUser Input("age"));
        height = Double.parseDouble(getUser Input("height in meters"));
        weight = Double.parseDouble(getUser Input("weight in kg"));

        bmi = calculateBMI(height, weight);
        bmi_status = determineBMIStatus(bmi);

        displayResult(name, age, height, weight, bmi, bmi_status);
    }

    public static String getUser Input(String prompt) {
        Scanner input = new Scanner(System.in);
        System.out.println("Enter your " + prompt + ": ");
        return input.next();
    }

    public static double calculateBMI(double height, double weight) {
        return weight / (height * height);
    }

    public static String determineBMIStatus(double bmi) {
        if (bmi < 18.5) {
            return "Underweight";
        } else if (bmi >= 18.5 && bmi <= 24.9) {
            return "Normal Weight";
        } else if (bmi >= 25 && bmi <= 29.9) {
            return "Overweight";
        } else {
            return "Obesity";
        }
    }

    public static void displayResult(String name, int age, double height, double weight, double bmi, String bmi_status) {
        System.out.println("\nThe Health Status for " + name );
        System.out.println("Age: " + age);
        System.out.println("Height: " + height + " meters");
        System.out.println("Weight: " + weight + " kg");
        System.out.println("BMI: " + bmi);
        System.out.println("Status: " + bmi_status);
    }
}

import java.util.Scanner;

public class assignment4Practice1 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        char repeat;

        do {
          System.out.println("This program calculates the average of three test scores.");

          double score1 = getScore(input, 1);
          double score2 = getScore(input, 2);
          double score3 = getScore(input, 3);

          double average = calculateAverage(score1, score2, score3);
          System.out.println("The average is " + average);

          System.out.print("Would you like to average another set of test scores? (Enter Y for yes or N for no): ");
          repeat = input.next().charAt(0);

        } while (repeat == 'Y' || repeat == 'y');

        System.out.println("Thank you for using the program!");
        input.close();
    }

    public static double getScore(Scanner input, int scoreNumber) {
        System.out.print("Enter score #" + scoreNumber + ": ");
        return input.nextDouble();
    }

    public static double calculateAverage(double score1, double score2, double score3) {
        return (score1 + score2 + score3) / 3;
    }
}
